-- SCRIPT PARA CRIAR USUÁRIO ADMIN (Supabase)
-- Execute este script no SQL Editor do Supabase para criar o usuário 'admin'
-- Senha definida: 123456
-- Email virtual: admin@sys.local (O sistema converte o login 'admin' para este email automaticamente)

-- 1. Garante que a extensão de criptografia esteja ativa
CREATE EXTENSION IF NOT EXISTS pgcrypto;

-- 2. Insere o usuário na tabela de autenticação (auth.users)
WITH new_user AS (
  INSERT INTO auth.users (
    instance_id,
    id,
    aud,
    role,
    email,
    encrypted_password,
    email_confirmed_at,
    raw_app_meta_data,
    raw_user_meta_data,
    created_at,
    updated_at,
    confirmation_token,
    email_change,
    email_change_token_new,
    recovery_token
  ) VALUES (
    '00000000-0000-0000-0000-000000000000',
    gen_random_uuid(),
    'authenticated',
    'authenticated',
    'admin@sys.local', -- IMPORTANTE: Domínio virtual usado pelo DataManager
    crypt('123456', gen_salt('bf')), -- Senha criptografada
    now(),
    '{"provider":"email","providers":["email"]}',
    '{"name": "Administrador", "nivel": "Admin", "mustChangePassword": false}',
    now(),
    now(),
    '',
    '',
    '',
    ''
  )
  RETURNING id
)
-- 3. Insere a identidade (Necessário para login funcionar corretamente)
INSERT INTO auth.identities (
  id,
  user_id,
  identity_data,
  provider,
  provider_id,
  last_sign_in_at,
  created_at,
  updated_at
)
SELECT
  gen_random_uuid(),
  id,
  format('{"sub":"%s","email":"admin@sys.local"}', id)::jsonb,
  'email',
  id::text,
  now(),
  now(),
  now()
FROM new_user;

-- NOTA: O perfil na tabela 'public.user_profiles' será criado automaticamente 
-- pelo Trigger definido no arquivo sql.txt. Se o trigger falhar, o usuário será criado sem perfil,
-- mas o DataManager tem um fallback para isso.
