-- 1. Estrutura do Banco de Dados (Correção de Colunas)

-- Habilita extensão para UUIDs
CREATE EXTENSION IF NOT EXISTS pgcrypto;

-- Cria a tabela se não existir (para instalações novas)
CREATE TABLE IF NOT EXISTS public.user_profiles (
  id UUID REFERENCES auth.users(id) ON DELETE CASCADE PRIMARY KEY,
  name TEXT NOT NULL,
  username TEXT,
  nivel TEXT DEFAULT 'Analista',
  must_change_password BOOLEAN DEFAULT true
);

-- CORREÇÃO: Adiciona a coluna must_change_password se a tabela já existir mas a coluna não
DO $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name = 'user_profiles' AND column_name = 'must_change_password') THEN
        ALTER TABLE public.user_profiles ADD COLUMN must_change_password BOOLEAN DEFAULT true;
    END IF;
END $$;

-- Tabela de Chamados
CREATE TABLE IF NOT EXISTS public.tickets (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES public.user_profiles(id),
  
  -- Informações do Cliente/Local
  customer_name TEXT,
  location_name TEXT,
  
  -- Detalhes Técnicos
  task_id TEXT,
  service_request TEXT,
  hostname TEXT,
  subject TEXT,
  
  -- Detalhes do Suporte
  analyst_name TEXT,
  support_start_time TIMESTAMPTZ,
  support_end_time TIMESTAMPTZ,
  description TEXT,
  analyst_action TEXT,
  
  -- Validadores e Peças
  is_due_call BOOLEAN DEFAULT false,
  used_acfs BOOLEAN DEFAULT false,
  has_ink_staining BOOLEAN DEFAULT false,
  part_replaced BOOLEAN DEFAULT false,
  part_description TEXT,
  
  -- Tags
  tag_vldd BOOLEAN DEFAULT false,
  tag_nlvdd BOOLEAN DEFAULT false,
  
  -- Validação Extra
  test_with_card BOOLEAN DEFAULT false,
  sic_withdrawal BOOLEAN DEFAULT false,
  sic_deposit BOOLEAN DEFAULT false,
  sic_sensors BOOLEAN DEFAULT false,
  sic_smart_power BOOLEAN DEFAULT false,
  
  -- Testemunha
  client_witness_name TEXT,
  client_witness_id TEXT,
  
  -- Validação
  validated_by TEXT,
  validated_at TIMESTAMPTZ,
  
  -- AI e Status
  ai_suggested_solution TEXT,
  status TEXT,
  priority TEXT,
  is_escalated BOOLEAN DEFAULT false,
  
  created_at TIMESTAMPTZ DEFAULT now()
);

-- 2. Row Level Security (RLS) - SEGURANÇA
ALTER TABLE public.user_profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.tickets ENABLE ROW LEVEL SECURITY;

-- Remove policies antigas e permissivas
DO $$ 
BEGIN
    DROP POLICY IF EXISTS "Enable all access for authenticated users" ON public.user_profiles;
    DROP POLICY IF EXISTS "Enable all access for authenticated users" ON public.tickets;
    DROP POLICY IF EXISTS "Users can view their own tickets" ON public.tickets;
    DROP POLICY IF EXISTS "Admins can view all tickets" ON public.tickets;
    DROP POLICY IF EXISTS "Users can insert their own tickets" ON public.tickets;
    DROP POLICY IF EXISTS "Users can update their own tickets" ON public.tickets;
END $$;

-- Políticas de Acesso (User Profiles)
-- Todos podem ler perfis (necessário para identificar nomes de usuários)
CREATE POLICY "Allow read access for authenticated users" ON public.user_profiles
FOR SELECT TO authenticated USING (true);

-- Apenas o próprio usuário ou trigger pode editar (Admin logicamente controlado via API, mas banco protege updates diretos)
CREATE POLICY "Users can update own profile" ON public.user_profiles
FOR UPDATE TO authenticated USING (auth.uid() = id);


-- Políticas de Acesso (Tickets)

-- 1. Analistas veem apenas seus próprios chamados
CREATE POLICY "Users can view their own tickets" ON public.tickets
FOR SELECT TO authenticated USING (
  auth.uid() = user_id
);

-- 2. Admins veem TUDO
CREATE POLICY "Admins can view all tickets" ON public.tickets
FOR SELECT TO authenticated USING (
  EXISTS (
    SELECT 1 FROM public.user_profiles 
    WHERE id = auth.uid() AND nivel = 'Admin'
  )
);

-- 3. Inserção permitida se o user_id for o próprio usuário
CREATE POLICY "Users can insert their own tickets" ON public.tickets
FOR INSERT TO authenticated WITH CHECK (
  auth.uid() = user_id
);

-- 4. Atualização (Admin edita qualquer um, Usuário edita o seu)
CREATE POLICY "Users and Admins can update tickets" ON public.tickets
FOR UPDATE TO authenticated USING (
  auth.uid() = user_id OR 
  EXISTS (
    SELECT 1 FROM public.user_profiles 
    WHERE id = auth.uid() AND nivel = 'Admin'
  )
);

-- 3. Trigger para Novos Usuários (Atualizado)
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS trigger AS $$
BEGIN
  INSERT INTO public.user_profiles (id, name, username, nivel, must_change_password)
  VALUES (
    new.id, 
    COALESCE(new.raw_user_meta_data->>'name', 'Usuário Novo'),
    split_part(new.email, '@', 1),
    COALESCE(new.raw_user_meta_data->>'nivel', 'Analista'),
    -- Garante que leia corretamente o booleano, padrão true se nulo
    COALESCE((new.raw_user_meta_data->>'mustChangePassword')::boolean, true)
  );
  RETURN new;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Recria o Trigger
DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;

CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE PROCEDURE public.handle_new_user();
