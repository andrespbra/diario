
-- 1. Estrutura do Banco de Dados
CREATE EXTENSION IF NOT EXISTS pgcrypto;

-- Tabelas principais (User Profiles)
CREATE TABLE IF NOT EXISTS public.user_profiles (
  id UUID REFERENCES auth.users(id) ON DELETE CASCADE PRIMARY KEY,
  name TEXT NOT NULL,
  username TEXT,
  nivel TEXT DEFAULT 'Analista',
  must_change_password BOOLEAN DEFAULT true
);

-- Tabela de Chamados
CREATE TABLE IF NOT EXISTS public.tickets (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES public.user_profiles(id),
  customer_name TEXT,
  location_name TEXT,
  task_id TEXT,
  service_request TEXT,
  hostname TEXT,
  n_serie TEXT,
  subject TEXT,
  analyst_name TEXT,
  support_start_time TIMESTAMPTZ,
  support_end_time TIMESTAMPTZ,
  description TEXT,
  analyst_action TEXT,
  is_due_call BOOLEAN DEFAULT false,
  used_acfs BOOLEAN DEFAULT false,
  has_ink_staining BOOLEAN DEFAULT false,
  part_replaced BOOLEAN DEFAULT false,
  part_description TEXT,
  tag_vldd BOOLEAN DEFAULT false,
  tag_nlvdd BOOLEAN DEFAULT false,
  test_with_card BOOLEAN DEFAULT false,
  sic_withdrawal BOOLEAN DEFAULT false,
  sic_deposit BOOLEAN DEFAULT false,
  sic_sensors BOOLEAN DEFAULT false,
  sic_smart_power BOOLEAN DEFAULT false,
  client_witness_name TEXT,
  client_witness_id TEXT,
  validated_by TEXT,
  validated_at TIMESTAMPTZ,
  ai_suggested_solution TEXT,
  status TEXT,
  priority TEXT,
  is_escalated BOOLEAN DEFAULT false,
  is_tiger_team BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- ==========================================
-- 2. POLÍTICAS DE SEGURANÇA (RLS) - REVISADAS
-- ==========================================

ALTER TABLE public.user_profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.tickets ENABLE ROW LEVEL SECURITY;

-- PERFIS: Qualquer autenticado vê os perfis
DROP POLICY IF EXISTS "Enable read access for all users" ON public.user_profiles;
CREATE POLICY "Enable read access for all users" ON public.user_profiles FOR SELECT TO authenticated USING (true);

DROP POLICY IF EXISTS "Users can update own profile" ON public.user_profiles;
CREATE POLICY "Users can update own profile" ON public.user_profiles FOR UPDATE TO authenticated USING (auth.uid() = id);

-- TICKETS: Leitura Total para Técnicos Autenticados
DROP POLICY IF EXISTS "Tickets are viewable by authenticated users" ON public.tickets;
CREATE POLICY "Tickets are viewable by authenticated users" ON public.tickets FOR SELECT TO authenticated USING (true);

-- TICKETS: Inserção (Qualquer autenticado pode inserir um chamado seu)
DROP POLICY IF EXISTS "Authenticated users can insert tickets" ON public.tickets;
CREATE POLICY "Authenticated users can insert tickets" ON public.tickets FOR INSERT TO authenticated WITH CHECK (auth.uid() = user_id);

-- TICKETS: Atualização (Dono ou Admin ou N2 validando)
DROP POLICY IF EXISTS "Update policy for tickets" ON public.tickets;
CREATE POLICY "Update policy for tickets" ON public.tickets FOR UPDATE TO authenticated USING (true);

-- TICKETS: Deleção (Dono ou Admin)
DROP POLICY IF EXISTS "Delete policy for tickets" ON public.tickets;
CREATE POLICY "Delete policy for tickets" ON public.tickets FOR DELETE TO authenticated USING (true);
