
-- 1. Estrutura do Banco de Dados
CREATE EXTENSION IF NOT EXISTS pgcrypto;

-- Tabelas principais (User Profiles)
CREATE TABLE IF NOT EXISTS public.user_profiles (
  id UUID REFERENCES auth.users(id) ON DELETE CASCADE PRIMARY KEY,
  name TEXT NOT NULL,
  username TEXT,
  nivel TEXT DEFAULT 'Analista',
  must_change_password BOOLEAN DEFAULT true
);

-- Tabela de Chamados
CREATE TABLE IF NOT EXISTS public.tickets (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES public.user_profiles(id),
  customer_name TEXT,
  location_name TEXT,
  task_id TEXT,
  service_request TEXT,
  hostname TEXT,
  n_serie TEXT,
  subject TEXT,
  analyst_name TEXT,
  support_start_time TIMESTAMPTZ,
  support_end_time TIMESTAMPTZ,
  description TEXT,
  analyst_action TEXT,
  is_due_call BOOLEAN DEFAULT false,
  used_acfs BOOLEAN DEFAULT false,
  has_ink_staining BOOLEAN DEFAULT false,
  part_replaced BOOLEAN DEFAULT false,
  part_description TEXT,
  tag_vldd BOOLEAN DEFAULT false,
  tag_nlvdd BOOLEAN DEFAULT false,
  test_with_card BOOLEAN DEFAULT false,
  sic_withdrawal BOOLEAN DEFAULT false,
  sic_deposit BOOLEAN DEFAULT false,
  sic_sensors BOOLEAN DEFAULT false,
  sic_smart_power BOOLEAN DEFAULT false,
  client_witness_name TEXT,
  client_witness_id TEXT,
  validated_by TEXT,
  validated_at TIMESTAMPTZ,
  ai_suggested_solution TEXT,
  status TEXT,
  priority TEXT,
  is_escalated BOOLEAN DEFAULT false,
  is_tiger_team BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- ==========================================
-- 2. POLÍTICAS DE SEGURANÇA (RLS) - REVISADAS
-- ==========================================

ALTER TABLE public.user_profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.tickets ENABLE ROW LEVEL SECURITY;

-- PERFIS: Qualquer autenticado vê os perfis (necessário para joins e validações)
DROP POLICY IF EXISTS "Perfis visíveis para autenticados" ON public.user_profiles;
CREATE POLICY "Perfis visíveis para autenticados" 
ON public.user_profiles FOR SELECT 
TO authenticated 
USING (true);

DROP POLICY IF EXISTS "Usuários atualizam próprio perfil" ON public.user_profiles;
CREATE POLICY "Usuários atualizam próprio perfil" 
ON public.user_profiles FOR UPDATE 
TO authenticated 
USING (auth.uid() = id);

-- TICKETS: Leitura Total (Para que a triagem N2 e Tiger Team funcione para todos)
DROP POLICY IF EXISTS "Tickets visíveis para todos os autenticados" ON public.tickets;
CREATE POLICY "Tickets visíveis para todos os autenticados" 
ON public.tickets FOR SELECT 
TO authenticated 
USING (true);

-- TICKETS: Inserção (Apenas o próprio usuário)
DROP POLICY IF EXISTS "Inserção de tickets próprios" ON public.tickets;
CREATE POLICY "Inserção de tickets próprios" 
ON public.tickets FOR INSERT 
TO authenticated 
WITH CHECK (auth.uid() = user_id);

-- TICKETS: Atualização (Dono, N2 em chamados escalonados ou Admin)
DROP POLICY IF EXISTS "Atualização de tickets autorizada" ON public.tickets;
CREATE POLICY "Atualização de tickets autorizada" 
ON public.tickets FOR UPDATE 
TO authenticated 
USING (
  auth.uid() = user_id OR 
  is_escalated = true OR 
  is_tiger_team = true OR
  EXISTS (SELECT 1 FROM public.user_profiles WHERE id = auth.uid() AND nivel = 'Admin')
);

-- TICKETS: Deleção (Dono ou Admin)
DROP POLICY IF EXISTS "Deleção de tickets autorizada" ON public.tickets;
CREATE POLICY "Deleção de tickets autorizada" 
ON public.tickets FOR DELETE 
TO authenticated 
USING (
  auth.uid() = user_id OR 
  EXISTS (SELECT 1 FROM public.user_profiles WHERE id = auth.uid() AND nivel = 'Admin')
);
