
-- 1. Estrutura do Banco de Dados (Correção de Colunas)

-- Habilita extensão para UUIDs
CREATE EXTENSION IF NOT EXISTS pgcrypto;

-- Cria a tabela se não existir
CREATE TABLE IF NOT EXISTS public.user_profiles (
  id UUID REFERENCES auth.users(id) ON DELETE CASCADE PRIMARY KEY,
  name TEXT NOT NULL,
  username TEXT,
  nivel TEXT DEFAULT 'Analista',
  must_change_password BOOLEAN DEFAULT true
);

-- Adiciona a coluna must_change_password se a tabela já existir mas a coluna não
DO $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name = 'user_profiles' AND column_name = 'must_change_password') THEN
        ALTER TABLE public.user_profiles ADD COLUMN must_change_password BOOLEAN DEFAULT true;
    END IF;
END $$;

-- Tabela de Chamados
CREATE TABLE IF NOT EXISTS public.tickets (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES public.user_profiles(id),
  
  -- Informações do Cliente/Local
  customer_name TEXT,
  location_name TEXT,
  
  -- Detalhes Técnicos
  task_id TEXT,
  service_request TEXT,
  hostname TEXT,
  n_serie TEXT, -- Número de Série
  subject TEXT,
  
  -- Detalhes do Suporte
  analyst_name TEXT,
  support_start_time TIMESTAMPTZ,
  support_end_time TIMESTAMPTZ,
  description TEXT,
  analyst_action TEXT,
  
  -- Validadores e Peças
  is_due_call BOOLEAN DEFAULT false,
  used_acfs BOOLEAN DEFAULT false,
  has_ink_staining BOOLEAN DEFAULT false,
  part_replaced BOOLEAN DEFAULT false,
  part_description TEXT,
  
  -- Tags
  tag_vldd BOOLEAN DEFAULT false,
  tag_nlvdd BOOLEAN DEFAULT false,
  
  -- Validação Extra
  test_with_card BOOLEAN DEFAULT false,
  sic_withdrawal BOOLEAN DEFAULT false,
  sic_deposit BOOLEAN DEFAULT false,
  sic_sensors BOOLEAN DEFAULT false,
  sic_smart_power BOOLEAN DEFAULT false,
  
  -- Testemunha
  client_witness_name TEXT,
  client_witness_id TEXT,
  
  -- Validação
  validated_by TEXT,
  validated_at TIMESTAMPTZ,
  
  -- AI e Status
  ai_suggested_solution TEXT,
  status TEXT,
  priority TEXT,
  is_escalated BOOLEAN DEFAULT false,
  is_tiger_team BOOLEAN DEFAULT false,
  
  created_at TIMESTAMPTZ DEFAULT now()
);

-- Garantia de colunas (Idempotente)
DO $$
BEGIN
    -- Coluna is_tiger_team
    IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name = 'tickets' AND column_name = 'is_tiger_team') THEN
        ALTER TABLE public.tickets ADD COLUMN is_tiger_team BOOLEAN DEFAULT false;
    END IF;
    
    -- Coluna n_serie
    IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name = 'tickets' AND column_name = 'n_serie') THEN
        ALTER TABLE public.tickets ADD COLUMN n_serie TEXT;
    END IF;

    -- Coluna subject (Caso falte)
    IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name = 'tickets' AND column_name = 'subject') THEN
        ALTER TABLE public.tickets ADD COLUMN subject TEXT;
    END IF;
END $$;
