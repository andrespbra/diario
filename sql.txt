
-- SCRIPT DE REPARO TOTAL - DIÁRIO DE BORDO
-- Execute este script no SQL Editor do Supabase para limpar erros de recursão e criar as tabelas

-- 1. Limpeza de Políticas Antigas (Evita Recursão Infinita)
DROP POLICY IF EXISTS "allow_all_read_profiles" ON public.user_profiles;
DROP POLICY IF EXISTS "allow_all_read_tickets" ON public.tickets;
DROP POLICY IF EXISTS "allow_auth_insert_tickets" ON public.tickets;
DROP POLICY IF EXISTS "allow_auth_update_tickets" ON public.tickets;
DROP POLICY IF EXISTS "Profiles are viewable by authenticated users" ON public.user_profiles;
DROP POLICY IF EXISTS "Tickets are viewable by authenticated users" ON public.tickets;
DROP POLICY IF EXISTS "Users can insert their own tickets" ON public.tickets;

-- 2. Garantir que as Tabelas existam
CREATE TABLE IF NOT EXISTS public.user_profiles (
    id UUID REFERENCES auth.users ON DELETE CASCADE PRIMARY KEY,
    name TEXT NOT NULL,
    username TEXT UNIQUE NOT NULL,
    nivel TEXT DEFAULT 'Analista',
    must_change_password BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.tickets (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    user_id UUID REFERENCES auth.users ON DELETE SET NULL,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    customer_name TEXT,
    location_name TEXT,
    task_id TEXT,
    service_request TEXT,
    hostname TEXT,
    n_serie TEXT,
    subject TEXT,
    analyst_name TEXT,
    support_start_time TEXT,
    support_end_time TEXT,
    description TEXT,
    analyst_action TEXT,
    is_due_call BOOLEAN DEFAULT FALSE,
    used_acfs BOOLEAN DEFAULT FALSE,
    has_ink_staining BOOLEAN DEFAULT FALSE,
    part_replaced BOOLEAN DEFAULT FALSE,
    part_description TEXT,
    tag_vldd BOOLEAN DEFAULT FALSE,
    tag_nlvdd BOOLEAN DEFAULT FALSE,
    test_with_card BOOLEAN DEFAULT FALSE,
    sic_withdrawal BOOLEAN DEFAULT FALSE,
    sic_deposit BOOLEAN DEFAULT FALSE,
    sic_sensors BOOLEAN DEFAULT FALSE,
    sic_smart_power BOOLEAN DEFAULT FALSE,
    client_witness_name TEXT,
    client_witness_id TEXT,
    status TEXT DEFAULT 'Aberto',
    priority TEXT DEFAULT 'Média',
    is_escalated BOOLEAN DEFAULT FALSE,
    is_tiger_team BOOLEAN DEFAULT FALSE,
    ai_suggested_solution TEXT,
    validated_by TEXT,
    validated_at TIMESTAMPTZ
);

-- 3. Habilitar Segurança (RLS)
ALTER TABLE public.user_profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.tickets ENABLE ROW LEVEL SECURITY;

-- 4. Novas Políticas SIMPLES (Sem recursão)
-- Permite que qualquer usuário logado veja os perfis e os tickets
CREATE POLICY "read_profiles" ON public.user_profiles FOR SELECT TO authenticated USING (true);
CREATE POLICY "read_tickets" ON public.tickets FOR SELECT TO authenticated USING (true);

-- Permite inserção de tickets por usuários logados
CREATE POLICY "insert_tickets" ON public.tickets FOR INSERT TO authenticated WITH CHECK (true);

-- Permite atualização de tickets por usuários logados
CREATE POLICY "update_tickets" ON public.tickets FOR UPDATE TO authenticated USING (true);

-- 5. Trigger de Perfil (Garante que o perfil exista ao logar)
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS trigger AS $$
BEGIN
  INSERT INTO public.user_profiles (id, name, username, nivel, must_change_password)
  VALUES (
    new.id, 
    COALESCE(new.raw_user_meta_data->>'name', split_part(new.email, '@', 1)), 
    COALESCE(new.raw_user_meta_data->>'username', split_part(new.email, '@', 1)), 
    COALESCE(new.raw_user_meta_data->>'nivel', 'Analista'), 
    TRUE
  )
  ON CONFLICT (id) DO UPDATE SET
    name = EXCLUDED.name,
    username = EXCLUDED.username;
  RETURN new;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE PROCEDURE public.handle_new_user();
