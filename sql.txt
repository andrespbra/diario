
-- 1. Estrutura do Banco de Dados
CREATE EXTENSION IF NOT EXISTS pgcrypto;

-- Tabelas principais (User Profiles)
CREATE TABLE IF NOT EXISTS public.user_profiles (
  id UUID REFERENCES auth.users(id) ON DELETE CASCADE PRIMARY KEY,
  name TEXT NOT NULL,
  username TEXT,
  nivel TEXT DEFAULT 'Analista',
  must_change_password BOOLEAN DEFAULT true
);

-- Tabela de Chamados
CREATE TABLE IF NOT EXISTS public.tickets (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES public.user_profiles(id),
  customer_name TEXT,
  location_name TEXT,
  task_id TEXT,
  service_request TEXT,
  hostname TEXT,
  n_serie TEXT,
  subject TEXT,
  analyst_name TEXT,
  support_start_time TIMESTAMPTZ,
  support_end_time TIMESTAMPTZ,
  description TEXT,
  analyst_action TEXT,
  is_due_call BOOLEAN DEFAULT false,
  used_acfs BOOLEAN DEFAULT false,
  has_ink_staining BOOLEAN DEFAULT false,
  part_replaced BOOLEAN DEFAULT false,
  part_description TEXT,
  tag_vldd BOOLEAN DEFAULT false,
  tag_nlvdd BOOLEAN DEFAULT false,
  test_with_card BOOLEAN DEFAULT false,
  sic_withdrawal BOOLEAN DEFAULT false,
  sic_deposit BOOLEAN DEFAULT false,
  sic_sensors BOOLEAN DEFAULT false,
  sic_smart_power BOOLEAN DEFAULT false,
  client_witness_name TEXT,
  client_witness_id TEXT,
  validated_by TEXT,
  validated_at TIMESTAMPTZ,
  ai_suggested_solution TEXT,
  status TEXT,
  priority TEXT,
  is_escalated BOOLEAN DEFAULT false,
  is_tiger_team BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- ==========================================
-- 2. POLÍTICAS DE SEGURANÇA (RLS) - CRUCIAL
-- ==========================================

-- Habilitar RLS nas tabelas
ALTER TABLE public.user_profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.tickets ENABLE ROW LEVEL SECURITY;

-- POLÍTICAS PARA USER_PROFILES
DROP POLICY IF EXISTS "Usuários podem ver o próprio perfil" ON public.user_profiles;
CREATE POLICY "Usuários podem ver o próprio perfil" 
ON public.user_profiles FOR SELECT 
TO authenticated 
USING (auth.uid() = id);

DROP POLICY IF EXISTS "Admins podem ver todos os perfis" ON public.user_profiles;
CREATE POLICY "Admins podem ver todos os perfis" 
ON public.user_profiles FOR SELECT 
TO authenticated 
USING (EXISTS (SELECT 1 FROM public.user_profiles WHERE id = auth.uid() AND nivel = 'Admin'));

-- POLÍTICAS PARA TICKETS
DROP POLICY IF EXISTS "Analistas veem seus tickets ou Tiger Team" ON public.tickets;
CREATE POLICY "Analistas veem seus tickets ou Tiger Team" 
ON public.tickets FOR SELECT 
TO authenticated 
USING (
  user_id = auth.uid() OR 
  is_tiger_team = true OR 
  is_escalated = true OR
  EXISTS (SELECT 1 FROM public.user_profiles WHERE id = auth.uid() AND nivel = 'Admin')
);

DROP POLICY IF EXISTS "Usuários podem inserir seus próprios tickets" ON public.tickets;
CREATE POLICY "Usuários podem inserir seus próprios tickets" 
ON public.tickets FOR INSERT 
TO authenticated 
WITH CHECK (user_id = auth.uid());

DROP POLICY IF EXISTS "Usuários podem atualizar seus próprios tickets" ON public.tickets;
CREATE POLICY "Usuários podem atualizar seus próprios tickets" 
ON public.tickets FOR UPDATE 
TO authenticated 
USING (
  user_id = auth.uid() OR 
  is_escalated = true OR
  EXISTS (SELECT 1 FROM public.user_profiles WHERE id = auth.uid() AND nivel = 'Admin')
);

DROP POLICY IF EXISTS "Apenas Admins ou Donos deletam" ON public.tickets;
CREATE POLICY "Apenas Admins ou Donos deletam" 
ON public.tickets FOR DELETE 
TO authenticated 
USING (
  user_id = auth.uid() OR 
  EXISTS (SELECT 1 FROM public.user_profiles WHERE id = auth.uid() AND nivel = 'Admin')
);
